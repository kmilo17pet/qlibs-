59 template<size_t order>
60 struct continuousTF {
61 real_t num[ order+1 ];
62 real_t den[ order+1 ];
63 continuousStates<order> states = {};
64 continuousTF( const real_t ( &numerator )[ order + 1 ], const real_t ( &denominator )[ order + 1 ] ) {
65 static_assert( order >= 1 , "Order should be greater than 0" );
66 for ( size_t i = 0; i <= order; ++i ) {
67 num[ i ] = numerator[ i ];
68 den[ i ] = denominator[ i ];
69 }
70 }
71 };
